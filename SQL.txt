CREATE TABLE Admin
(
  Admin_ID INT(6) AUTO_INCREMENT PRIMARY KEY,
  User_Name VARCHAR(50) NOT NULL,
  Password VARCHAR(50) NOT NULL
);

CREATE TABLE Applicant
(
  User_ID INT(6) AUTO_INCREMENT PRIMARY KEY,
  User_Name VARCHAR(50) NOT NULL,
  Password VARCHAR(50) NOT NULL,
  Email VARCHAR(50) NOT NULL,
  UNIQUE (Email),
  City VARCHAR(50) NOT NULL,
  NID INT NOT NULL
);

CREATE TABLE Jobs
(
  Job_ID INT(6) AUTO_INCREMENT PRIMARY KEY,
  Job_Title VARCHAR(100) NOT NULL,
  Job_Description VARCHAR(500) NOT NULL,
  Salary INT NOT NULL,
  Location VARCHAR(50) NOT NULL,
  Company_Name VARCHAR(100) NOT NULL,
  Qualification VARCHAR(100) NOT NULL,
  Interview_Date DATE NOT NULL,
  Interview_Time TIME NOT NULL
);

CREATE TABLE Posted_Jobs
(
  Admin_ID INT NOT NULL,
  Job_ID INT NOT NULL,
  PRIMARY KEY (Admin_ID, Job_ID),
  FOREIGN KEY (Admin_ID) REFERENCES Admin(Admin_ID),
  FOREIGN KEY (Job_ID) REFERENCES Jobs(Job_ID)
);

CREATE TABLE Apply
(
  Admin_ID INT(6) AUTO_INCREMENT PRIMARY KEY,
  Status INT NOT NULL,
  Notify_user INT NOT NULL,
  Notify_admin INT NOT NULL,
  User_ID INT NOT NULL,
  Job_ID INT NOT NULL
);

CREATE TABLE Temp
(
  User_ID INT NOT NULL,
  Password VARCHAR(50) NOT NULL, 
  City VARCHAR(50) NOT NULL,
  PRIMARY KEY (User_ID)
)


CREATE TABLE `form_data` (
  `User_ID` int(6) NOT NULL,
  `file_name` varchar(255) COLLATE utf8_unicode_ci NOT NULL,
  `submitted_on` datetime NOT NULL DEFAULT current_timestamp(),
  PRIMARY KEY (`User_ID`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COLLATE=utf8_unicode_ci;



CREATE TABLE `form_data` (
  `User_ID` int(6) NOT NULL,
  `file_name` varchar(255) COLLATE utf8_unicode_ci NOT NULL,
  `submitted_on` datetime NOT NULL DEFAULT current_timestamp(),
  PRIMARY KEY (`User_ID`)
) ENGINE=MyISAM DEFAULT CHARSET=latin1;


CREATE TABLE IF NOT EXISTS `pdf_data` (
 `User_ID` int(6) NOT NULL,
  `username` varchar(50) NOT NULL,
  `filename` varchar(255) NOT NULL,
  PRIMARY KEY (`id`)
) ENGINE=MyISAM DEFAULT CHARSET=latin1;




CREATE TABLE files(
  User_ID int (6),
  file_name varchar(255) NOT NULL,
  file_path varchar(255) NOT NULL,
  PRIMARY KEY (User_ID)
)
